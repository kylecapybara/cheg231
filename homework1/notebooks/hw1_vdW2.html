import numpy as np
import scipy.optimize as opt
import matplotlib.pyplot as plt

a = 0.3634 
b = 4.236e-5 
R = 8.314 
P = 5e5

def vdWT(V, T):
    return (R * T) - ((P + (a / V**2)) * (V - b))

temperatures = np.linspace(150,450,100)

vdw_volumes = np.array([opt.fsolve(vdWT, R*T/P, args=(T))[0] for T in temperatures])

ideals = R * temperatures / P

plt.plot(temperatures, vdw_volumes, label='van der Waals equation')
plt.plot(temperatures, ideals, label='Ideal Gas equation')
plt.xlim(150,450)
plt.xlabel('Temperature (K)')
plt.ylabel('Molar Volume (m^3/mol)')
plt.title('van der Waals vs. Ideal Gas for $CO_2$ molar volume')
plt.tight_layout()
plt.legend()
