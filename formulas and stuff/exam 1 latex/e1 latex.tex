\subsection*{mass balance, difference form}

\[
    \Delta M = \Delta M_{\text{in}} - \Delta M_{\text{out}}
\]

\subsection*{mass balance, differential form}

\[
    \frac{dM}{dt} = \dot{M_{\text{in}}} - \dot{M_{\text{out}}}
\]
